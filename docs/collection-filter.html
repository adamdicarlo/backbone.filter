Backbone.CollectionFilter

Returns a Backbone.Collection containing models filtered from an existing Backbone.Collection

myCollection = new MyCollection([
  { name: "John" },
  { name: "Joe" },
  { name: "Jim" }
])

filter = new MyFilter
newCollection = filter.apply(myCollection, { name: 'John' })

Provide options to the filter by passing them in during initialization:

opts =
  optionA: 'a'
  optionB: 'b'

filter = new MyFilter(opts)

Note that these options are whitelisted against the defaults attribute

 
class Backbone.CollectionFilter

  defaults:
    searchAttributes: []

  constructor: (opts = {}) ->
    @options ||= {}
    for key in _.keys(@defaults)
      @options[key] = opts[key] || @defaults[key]
¶

Filters should override the comparator method to provide their own comparators. Implementations should take a model in the context of the current filter.

 
  comparator: (model) ->
    for attribute in @options.searchAttributes
      return true if model.get(attribute) == @query
    false
¶

The match method is used to return a collection containing all models that pass the comparator() test

 
  apply: (collection, @query) ->
    models = collection.filter @comparator, @
    new collection.constructor models
¶

Patch Backbone's Collection.filter method to support collection filters.

 
do (Backbone) ->
  filterFunc = Backbone.Collection.prototype.filter

  Backbone.Collection.prototype.filter = (meth, args...) ->
    if meth.comparator? # Instantiated filter, cool.
      meth.apply(@, args[0])
    else if meth.prototype.comparator? # Filter, cool.
      query = args.pop()
      filter = new meth args[0]
      results = filter.apply(@, query)
    else
      filterFunc.apply(@, [meth].concat(args...))
